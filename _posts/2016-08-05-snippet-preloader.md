      $('.filter a').click(function() {
        $('.filter .current').removeClass('current');
        $(this).addClass('current');

        $('.grid').isotope({
          filter: selector
        });
        return false;
      });

        helpers: {
          overlay: {
            locked: false
          }

      });

        $('.loader').delay(500).fadeOut('slow').promise().done(function() {
          $('.wrapper').fadeIn('slow');
